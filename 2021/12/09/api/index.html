模拟常用函数

发表于 2021-12-09 分类于 前端

记录开发中遇过的工具函数或是简单模拟函数的实现

useClickAway

useClickAway：来自[ahooks](https://ahooks.js.org/)库，用于监听目标元素外的点击事件。

举个例子：指定点击某元素之外区域可触发log（类似局部蒙版效果）

/**
 * 指定：targets以外的DOM元素都能触发callback
 * @param {fn} callback 触发的事件函数
 * @param {array<dom>|dom} targets DOM节点或DOM节点数组
 * @param {string} eventName  绑定的事件类型
 */
function useClickAway(callback, targets, eventName) {
  eventName = eventName || "click"; // 默认点击事件;
  targets = Array.isArray(targets) ? targets : [targets]; // 转化成数组统一处理

  function wrappedEvent(event) {
    // 添加事件拦截逻辑，如果event包含了targets中的成员
    if (
      targets.some(tEle => getTargetElement(tEle).contains(event.target))
    ) {
      return;
    }
    callback(event);
  };
  document.addEventListener(eventName, wrappedEvent);
  // 抛出解绑事件
  return (function remove() {
    document.removeEventListener(eventName, wrappedEvent)
  });
};

/**
 * 获取真实的DOM节点, element可能是对象，方法
 * @param {any} element 
 */
function getTargetElement(element) {
  if (!element) return window;
  if (typeof element === "function") {
    return element();
  }
  if ("current" in element) { // 这里在doc.getElement时可以忽略
    return element.current;
  }
  return element;
};
// 使用：
const rmEventListen = useClickAway(
  function (event) {
    console.log("hahaha...")
  },
  [
    document.querySelector(".area-1"),
    document.querySelector("h2")
  ]
)

call

Function.prototype.useCall = function (context) {
  var that = context || window;
  var _arguments = [...arguments].slice(1);

  that.fnc = this;
  var result = that.fnc(..._arguments);
  delete that.fnc;
  return result;
};

apply

Function.prototype.useApply = function (context, args) {
  var that = context || window;
  that.fnc = this;
  var result = that.fnc(...args);
  delete that.fnc;
  return result;
};

bind

Function.prototype.useBind = function (context) {
  var that = context || window;
  var fnc = this;
  var firstArgs = [...arguments].slice(1);
  return function () {
    return fnc.apply(that, firstArgs.concat(arguments));
  };
};

new

function myNew(fn) {
  var obj = new Object({});
  obj.__proto__ = fn.prototype;

  var args = [...arguments].slice(1);
  fn.apply(obj, args);
  return obj;
};
