useClickAway

useClickAway：来自[ahooks](https://ahooks.js.org/)库，用于监听目标元素外的点击事件。

举个例子：指定点击某元素之外区域可触发log（类似局部蒙版效果）

/**
 * 指定：targets以外的DOM元素都能触发callback
 * @param {fn} callback 触发的事件函数
 * @param {array<dom>|dom} targets DOM节点或DOM节点数组
 * @param {string} eventName  绑定的事件类型
 */
function useClickAway(callback, targets, eventName) {
  eventName = eventName || "click"; // 默认点击事件;
  targets = Array.isArray(targets) ? targets : [targets]; // 转化成数组统一处理

  function wrappedEvent(event) {
    // 添加事件拦截逻辑，如果event包含了targets中的成员
    if (
      targets.some(tEle => getTargetElement(tEle).contains(event.target))
    ) {
      return;
    }
    callback(event);
  };
  document.addEventListener(eventName, wrappedEvent);
  // 抛出解绑事件
  return (function remove() {
    document.removeEventListener(eventName, wrappedEvent)
  });
};

/**
 * 获取真实的DOM节点, element可能是对象，方法
 * @param {any} element 
 */
function getTargetElement(element) {
  if (!element) return window;
  if (typeof element === "function") {
    return element();
  }
  if ("current" in element) { // 这里在doc.getElement时可以忽略
    return element.current;
  }
  return element;
};
// 使用：
const rmEventListen = useClickAway(

    console.log("hahaha...")
  },
  [
    document.querySelector(".area-1"),
    document.querySelector("h2")
  ]
)

call

Function.prototype.useCall = function (context) {
  var that = context || window;
  var _arguments = [...arguments].slice(1);

  that.fnc = this;
  var result = that.fnc(..._arguments);
  delete that.fnc;
  return result;
};

apply

Function.prototype.useApply = function (context, args) {
  var that = context || window;
  that.fnc = this;
  var result = that.fnc(...args);
  delete that.fnc;
  return result;
};

bind

Function.prototype.useBind = function (context) {
  var that = context || window;
  var fnc = this;
  var firstArgs = [...arguments].slice(1);
  return function () {
    return fnc.apply(that, firstArgs.concat(arguments));
  };
};

function myNew(fn) {

  obj.__proto__ = fn.prototype;

  return obj;

debounce

短时间内多次触发，每当触发一次，重新计时，直到期限时间结束后开始执行

function debounce(callback,delay) {
  let timer;
  return (
    function () {
      if (timer) {
        clearTimeout(timer)
      } 
      const args = arguments;
      const context = this;
      timer = setTimeout(() => {
        callback.apply(context,args);
      },delay);
    }
  )
};

throttle

多次触发，只有当前函数执行完后，才会开始执行下一次函数

function throttle(fn,delay) {
  let timer;
  return (
    function () {
      if (timer) return;
      timer = setTimeout(() => {
        fn.apply(this,arguments);
        clearTimeout(timer);
      })
    }
  )
};

/**

 * @param data: 需判断的目标数据

judgeDataType = data => {
  const type = Object.prototype.toString.call(data);
  switch (type) {
    case '[object Object]':
      return 'object';
    case '[object String]':
      return 'string';
    case '[object Array]':
      return 'array';
    case '[object Undefined]':
      return 'undefined';
    case '[object Function]':
      return 'function';
    case '[object Symbol]':
      return 'symbol';
    case '[object Null]':
      return 'null';
    case '[object RegExp]':
      return 'regexp';
    case '[object Number]':
      return 'number';
    case 'object Boolean':
      return 'boolean';
    default: return '';
  }
}

关于parseInt

parseInt(string, radix)中的入参radix是2-36 之间的整数,表示被解析字符串的基数。

而如果radix 是 undefined、0 或未指定的，则会根据基数做假定，或十六进制，或八进制，或十进制。

简言之，假设radix = 2，那么string就是一个二进制的数。

parseInt(1/0, 18); // NaN
parseInt(1/0, 19); // 18 
// detail: 1/0 === 'Infinity'  0123456789 a(10) b c d e f g h i(18)
parseInt(1/0, 19) === > parseInt('i',19);
[1,2,3].map(parseInt); // [1, NaN, NaN]

另外，parseInt('2',2) 这样的语句是会返回NaN的，因为解析不了（毕竟二进制数怎么会有2呢？）；

更不必说parseInt('3',2)了。

全等和等于

“==”允许在相等比较中进行强制类型转换，而“===”不允许。

“==”:

  1. 如果存在number类型，则都转number类型比较
  2. 如果存在boolean类型，则将boolean转number，然后将另一位转number

// 分发iframe事件
const dispatchIframeEvent = () => {
  divRef.current.contentDocument.body.removeEventListener('mousedown', iframeMouseDown)
  divRef.current.contentDocument.body.removeEventListener('mousemove', iframeMouseMove)
  divRef.current.contentDocument.body.removeEventListener('mouseup', iframeMouseUp);

  divRef.current.contentDocument.body.addEventListener('mousedown', iframeMouseDown)
  divRef.current.contentDocument.body.addEventListener('mousemove', iframeMouseMove)
  divRef.current.contentDocument.body.addEventListener('mouseup', iframeMouseUp);
}

const formatEvent = (e) => {
  const { left, top } = divRef.current.getBoundingClientRect();
  return ({
    clientX: e.clientX + left,
    clientY: e.clientY + top,
    target: e.target,
    // 这里的view，必须使用e.view且view属性必须存在
    view: e.view,
    screenX: e.screenX,
    screenY: e.screenY,
    button: e.button,
    bubbles: true,
  })
}

const iframeMouseDown = useCallback((e) => {
  const event = formatEvent(e);
  divRef.current.dispatchEvent(new MouseEvent('mousedown', event));
}, []);

const iframeMouseMove = useCallback((e) => {
  const event = formatEvent(e)
  divRef.current.dispatchEvent(new MouseEvent('mousemove', event));
}, []);

const iframeMouseUp = useCallback((e) => {
  const event = formatEvent(e);
  divRef.current.dispatchEvent(new MouseEvent('mouseup', event));
}, []);

DOM元素的id属性也是变量

由于浏览器演进过程中遗留历史问题，在创建带有id的DOM元素时，也会同时创建同名的全局变量
因此，将会出现以下情况： 
<div id="msg">
console.log(msg) // <div id="msg">
if (typeof msg === "undefined") { // 不会执行
  alert(222)
}

浏览器下for,forEach性能对比

参考： https://github.com/advanced-frontend/daily-interview-question/issues/121

简言之，就是forEach中存在自己维护的上下文环境，会额外消耗内存，且有诸多函数的存在。 需要判断 callback是否是函数，变量隐式转化等等
在10万这个级别下， forEach 的性能是 for的十倍 
for: 2.263ms
forEach: 0.254ms

在100万这个量级下， forEach 的性能是和for的一致

for: 2.844ms
forEach: 2.652ms

在1000万级以上的量级上 ， forEach 的性能远远低于for的性能

for: 8.422ms
forEach: 30.328m

箭头函数和普通函数的区别

1. 函数体内的 this 对象，就是定义时所在的作用域中的 this 值，而不是使用时所在的对象。this 相当于一个普通变量会向作用域链中查询结果，同时定义时所在对象也并不等于所在对象中的 this 值。
2. 不可以使用 arguments 对象，该对象在函数体内不存在。如果要用，可以用 rest 参数代替。
3. 不可以使用 yield 命令，因此箭头函数不能用作 Generator 函数。
4. 不可以使用 new 命令，因为：
  4.1 没有自己的 this，无法调用 call，apply。
  4.2 没有 prototype 属性 ，而 new 命令在执行时需要将构造函数的 prototype 赋值给新的对象的 __proto__

new 过程大致是这样的：

function newFunc(father, ...rest) {
  var result = {};
  result.__proto__ = father.prototype;
  var result2 = father.apply(result, rest);
  if (
    (typeof result2 === 'object' || typeof result2 === 'function') &&
    result2 !== null
  ) {
    return result2;
  }
  return result;
}
1. 创建一个空对象，并且 this 变量引用该对象，// let target = {};
2. 继承了函数的原型。// target.proto = func.prototype;
3. 属性和方法被加入到 this 引用的对象中。并执行了该函数func// func.call(target);
4. 新创建的对象由 this 所引用，并且最后隐式的返回 this ，如果func.call(target)返回的res是个对象或者function 就返回它

获取当前浏览器页面的一些基本信息

  • 环境信息

    • url: window.location.href （当前页面）
    • ua: window.navigator.userAgent （ userAgent信息，包含操作系统和浏览器的类型、版本等 ）
    • token信息

  • 性能信息

    • ​ window.performance

window.performance 下的timing属性记录了重定向，DNS解析，TCP连接，SSL连接，资源加载，数据传输，TTFB网络请求耗时的相关起止时间戳。

function listenerError() {
    window.addEventListener('error', (event) => {
        if (event.target.localName) {
            console.log('这是资源错误', event);
        }
        else {
            console.log('这是代码错误', event);
        }
    }, true)
}
// window.addEventListener('error') 比window.onError先执行，它可以全局捕获资源加载错误，但不能通过返回true的方式阻止执行默认事件处理函数

  function listenerPromiseError() {
